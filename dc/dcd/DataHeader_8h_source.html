161 template <typename T, typename... Targs>
162 constexpr T String2(char c, Targs... Fargs)
163 {
164  // number of arguments has either to match the size of the type, or the last element is treated
165  // as '0' if missing
166  // TODO: this is a bit of a question, do we allow the number of characters to
167  // be smaller than the size of the type and just pad with 0 like in the case
168  // of a char array argument?
169  static_assert(sizeof...(Targs) == sizeof(T) + 1 ||
170  sizeof...(Targs) == sizeof(T),
171  "number of arguments does not match the uint type width"
172  );
173  return Internal::String2__<T>(c, Fargs...);
174 }

180 template <typename T, std::size_t N, std::size_t pos = 0, bool suppressAssert = false>
181 constexpr T String2(const char (&str)[N])
182 {
183  static_assert(std::is_integral<T>::value, "Non integral types not compatible with String2<type>");
184  static_assert(N >= pos, "Position is larger than the length of string");
185  static_assert(suppressAssert || N - pos <= sizeof(T) + 1,
186  "String parameter is longer than the size of the data type"
187  );
188  return((T) str[0+pos] |
189  (str[0+pos] && sizeof(T) >= 2 ? ((T) str[1+pos] << (sizeof(T) >= 2 ? 8 : 0) |
190  (str[1+pos] && sizeof(T) >= 4 ? ((T) str[2+pos] << (sizeof(T) >= 4 ? 16 : 0) |
191  (str[2+pos] && sizeof(T) >= 4 ? ((T) str[3+pos] << (sizeof(T) >= 4 ? 24 : 0) |
192  (str[3+pos] && sizeof(T) >= 8 ? ((T) str[4+pos] << (sizeof(T) >= 8 ? 32 : 0) |
193  (str[4+pos] && sizeof(T) >= 8 ? ((T) str[5+pos] << (sizeof(T) >= 8 ? 40 : 0) |
194  (str[5+pos] && sizeof(T) >= 8 ? ((T) str[6+pos] << (sizeof(T) >= 8 ? 48 : 0) |
195  (str[6+pos] && sizeof(T) >= 8 ? ((T) str[7+pos] << (sizeof(T) >= 8 ? 56 : 0) )
196  : 0)) : 0)) : 0)) : 0)) : 0)) : 0)) : 0));
197 }

280  void runtimeInit(const char* string, short length = -1) {
281  char* target = str;
282  char* targetEnd = target;
283  if (length >= 0 && length < (int)N) targetEnd += length;
284  else targetEnd += N;
285  const char* source = string;
286  for ( ; source != nullptr && target < targetEnd && *source !=0; ++target, ++source) *target = *source;
287  targetEnd = str + N;
288  for ( ; target < targetEnd; ++target) *target = 0;
289  // require the string to be not longer than the descriptor size
290  assert(source != nullptr && (*source == 0 || (length >= 0 && length <= (int)N)));
291  }

294  bool operator!=(const Descriptor& other) const {return not this->operator==(other);}
