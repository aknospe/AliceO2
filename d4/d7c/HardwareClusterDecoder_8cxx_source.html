29 {
30  int nRowClusters[Constants::MAXSECTOR][Constants::MAXGLOBALPADROW] = {0};
31  ClusterNativeContainer* containerRowCluster[Constants::MAXSECTOR][Constants::MAXGLOBALPADROW] = {nullptr};
32  Mapper& mapper = Mapper::instance();
33  int numberOfOutputContainers = 0;
34  for (int loop = 0;loop < 2;loop++)
35  {
36  for (int i = 0;i < inputClusters.size();i++)
37  {
38  for (int j = 0;j < inputClusters[i].second;j++)
39  {
40  const char* tmpPtr = reinterpret_cast<const char*> (inputClusters[i].first);
41  tmpPtr += j * 8192; //TODO: FIXME: Compute correct offset based on the size of the actual packet in the RDH
42  const ClusterHardwareContainer& cont = *(reinterpret_cast<const ClusterHardwareContainer*> (tmpPtr));
43  const CRU cru(cont.mCRU);
44  const Sector sector = cru.sector();
45  const PadRegionInfo& region = mapper.getPadRegionInfo(cru.region());
46  const int rowOffset = region.getGlobalRowOffset();
47 
48  for (int k = 0;k < cont.mNumberOfClusters;k++)
49  {
50  const int padRowGlobal = rowOffset + cont.mClusters[k].mRow;
51  int& nCls = nRowClusters[sector][padRowGlobal];
52  if (loop == 1)
53  {
54  //Fill cluster in the respective output buffer
55  const ClusterHardware& cIn = cont.mClusters[k];
56  ClusterNative& cOut = containerRowCluster[sector][padRowGlobal]->mClusters[nCls];
57  float pad = cIn.getPad();
58  cOut.setPad(pad);
59  cOut.setTimeFlags(cIn.getTimeLocal() + cont.mTimeBinOffset, cIn.mFlags);
60  cOut.setSigmaPad2(cIn.getSigmaPad2());
61  cOut.setSigmaTime2(cIn.getSigmaTime2());
62  cOut.mQMax = cIn.mQMax;
63  cOut.mQTot = cIn.mQTot;
64  mIntegrator.integrateCluster(sector, padRowGlobal, pad, cIn.mQTot);
65  }
66  else
67  {
68  //Count how many output buffers we need (and how large they are below)
69  if (nCls == 0) numberOfOutputContainers++;
70  }
71  nCls++;
72  }
73  }
74  }
75  if (loop == 1)
76  {
77  //We are done with filling the buffers, sort all output buffers
78  for (int i = 0;i < outputClusters.size();i++)
79  {
80  auto& cl = outputClusters[i].mClusters;
81  std::sort(cl.data(), cl.data() + cl.size(), [](const auto& a, const auto& b) {
82  if (a.getTimePacked() != b.getTimePacked()) return(a.getTimePacked() < b.getTimePacked());
83  return(a.mPadPacked < b.mPadPacked);
84  });
85  }
86  }
87  else
88  {
89  //Now we know the size of all output buffers, allocate them
90  outputClusters.resize(numberOfOutputContainers);
91  numberOfOutputContainers = 0;
92  for (int i = 0;i < Constants::MAXSECTOR;i++)
93  {
94  for (int j = 0;j < Constants::MAXGLOBALPADROW;j++)
95  {
96  if (nRowClusters[i][j] == 0) continue;
97  outputClusters[numberOfOutputContainers].mClusters.resize(nRowClusters[i][j]);
98  outputClusters[numberOfOutputContainers].mSector = i;
99  outputClusters[numberOfOutputContainers].mGlobalPadRow = j;
100  containerRowCluster[i][j] = &outputClusters[numberOfOutputContainers++];
101  mIntegrator.initRow(i, j);
102  }
103  }
104  memset(nRowClusters, 0, sizeof(nRowClusters));
105  }
106  }
107  return(0);
