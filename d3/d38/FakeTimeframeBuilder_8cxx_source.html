56 std::unique_ptr<char[]> fakeTimeframeGenerator(std::vector<FakeTimeframeSpec> &specs, std::size_t &totalSize) {
57  // Calculate the total size of your timeframe. This is
58  // given by:
59  // - N*The size of the data header (this should actually depend on the
60  // kind of data as different dataDescriptions will probably have
61  // different headers).
62  // - Sum_N(The size of the buffer_i)
63  // - The size of the index header
64  // - N*sizeof(dataheader)
65  // Assuming all the data header
66  size_t sizeOfHeaders = specs.size()*sizeof(DataHeader);
67  size_t sizeOfBuffers = 0;
68  for (auto && spec : specs) {
69  sizeOfBuffers += spec.bufferSize;
70  }
71  size_t sizeOfIndexHeader = sizeof(DataHeader);
72  size_t sizeOfIndex = sizeof(IndexElement)*specs.size();
73  totalSize = sizeOfHeaders + sizeOfBuffers + sizeOfIndexHeader + sizeOfIndex;
74 
75  // Add the actual - data
76  auto buffer = std::make_unique<char[]>(totalSize);
77  char *bi = buffer.get();
78  std::vector<IndexElement> headers;
79  int count = 0;
80  for (auto &&spec : specs) {
81  IndexElement el;
82  el.first.dataDescription = lookupDataDescription(spec.dataDescription);
83  el.first.dataOrigin = lookupDataOrigin(spec.origin);
84  el.first.payloadSize = spec.bufferSize;
85  el.first.headerSize = sizeof(el.first);
86  el.second = count++;
87  // Let's zero at least the header...
88  memset(bi, 0, sizeof(el.first));
89  memcpy(bi, &el, sizeof(el.first));
90  headers.push_back(el);
91  bi += sizeof(el.first);
92  spec.bufferFiller(bi, spec.bufferSize);
93  bi += spec.bufferSize;
94  }
95 
96  // Add the index
97  DataHeader index;
98  index.dataDescription = DataDescription("TIMEFRAMEINDEX");
99  index.dataOrigin = DataOrigin("FKE");
100  index.headerSize = sizeOfIndexHeader;
101  index.payloadSize = sizeOfIndex;
102  memcpy(bi, &index, sizeof(index));
103  memcpy(bi+sizeof(index), headers.data(), headers.size() * sizeof(IndexElement));
104  return std::move(buffer);
105 }
