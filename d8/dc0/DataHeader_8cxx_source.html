236 void o2::Header::hexDump (const char* desc, const void* voidaddr, size_t len, size_t max)
237 {
238  size_t i;
239  unsigned char buff[17]; // stores the ASCII data
240  memset(&buff[0],'\0',17);
241  const byte* addr = reinterpret_cast<const byte*>(voidaddr);
242 
243  // Output description if given.
244  if (desc != nullptr)
245  printf ("%s, ", desc);
246  printf("%zu bytes:", len);
247  if (max>0 && len>max) {
248  len = max; //limit the output if requested
249  printf(" output limited to %zu bytes\n", len);
250  } else {
251  printf("\n");
252  }
253 
254  // In case of null pointer addr
255  if (addr==nullptr) {printf(" nullptr, size: %zu\n", len); return;}
256 
257  // Process every byte in the data.
258  for (i = 0; i < len; i++) {
259  // Multiple of 16 means new line (with line offset).
260  if ((i % 16) == 0) {
261  // Just don't print ASCII for the zeroth line.
262  if (i != 0)
263  printf (" %s\n", buff);
264 
265  // Output the offset.
266  //printf (" %04x ", i);
267  printf (" %p ", &addr[i]);
268  }
269 
270  // Now the hex code for the specific character.
271  printf (" %02x", addr[i]);
272 
273  // And store a printable ASCII character for later.
274  if ((addr[i] < 0x20) || (addr[i] > 0x7e))
275  buff[i % 16] = '.';
276  else
277  buff[i % 16] = addr[i];
278  buff[(i % 16) + 1] = '\0';
279  fflush(stdout);
280  }
281 
282  // Pad out last line if not exactly 16 characters.
283  while ((i % 16) != 0) {
284  printf (" ");
285  fflush(stdout);
286  i++;
287  }
288 
289  // And print the final ASCII bit.
290  printf (" %s\n", buff);
291  fflush(stdout);
292 }
