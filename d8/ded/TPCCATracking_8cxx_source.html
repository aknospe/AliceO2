66 int TPCCATracking::runTracking(TChain* inputClustersChain, const std::vector<o2::TPC::Cluster>* inputClustersArray, std::vector<TrackTPC>* outputTracks) {
67  if (mTrackingCAO2Interface == nullptr) return (1);
68  if (inputClustersChain && inputClustersArray) {
69  LOG(FATAL) << "Internal error, must not pass in both TChain and TClonesArray of clusters\n";
70  }
71 
72  int retVal = 0;
73  const static ParameterDetector &detParam = ParameterDetector::defaultInstance();
74  const static ParameterGas &gasParam = ParameterGas::defaultInstance();
75  const static ParameterElectronics &elParam = ParameterElectronics::defaultInstance();
76 
77  const AliHLTTPCGMMergedTrack* tracks;
78  int nTracks;
79  const unsigned int* trackClusterIDs;
80 
81  std::vector<Cluster> clusterCache;
82 
83  int nClusters[Sector::MAXSECTOR] = {};
84  int nClustersConverted = 0;
85  int nClustersTotal = 0;
86  int numChunks;
87  if (inputClustersChain) {
88  inputClustersChain->SetBranchAddress("TPCClusterHW", &inputClustersArray);
89  numChunks = inputClustersChain->GetEntries();
90  } else {
91  numChunks = 1;
92  }
93 
94  for (int iChunk = 0;iChunk < numChunks;iChunk++) {
95  if (inputClustersChain) {
96  inputClustersChain->GetEntry(iChunk);
97  }
98  for (const auto& cluster : *inputClustersArray) {
99  const Sector sector = CRU(cluster.getCRU()).sector();
100  nClusters[sector.getSector()]++;
101  }
102  nClustersTotal += inputClustersArray->size();
103  if (inputClustersChain) clusterCache.resize(nClustersTotal);
104  for (int i = 0; i < Sector::MAXSECTOR; i++) {
105  if (iChunk == 0) {
106  mClusterData[i].StartReading(i, nClusters[i]);
107  } else {
108  mClusterData[i].Allocate(mClusterData[i].NumberOfClusters() + nClusters[i]);
109  }
110  }
111  for (Int_t icluster = 0; icluster < inputClustersArray->size(); ++icluster) {
112  const auto& cluster = (*inputClustersArray)[icluster];
113  const CRU cru(cluster.getCRU());
114  const Sector sector = cru.sector();
115  AliHLTTPCCAClusterData& cd = mClusterData[sector.getSector()];
116  AliHLTTPCCAClusterData::Data& hltCluster = cd.Clusters()[cd.NumberOfClusters()];
117 
118  // ===| mapper |==============================================================
119  Mapper& mapper = Mapper::instance();
120  const PadRegionInfo& region = mapper.getPadRegionInfo(cru.region());
121  const int rowInSector = cluster.getRow() + region.getGlobalRowOffset();
122  const float padY = cluster.getPadMean();
123  const int padNumber = int(padY);
124  const GlobalPadNumber pad = mapper.globalPadNumber(PadPos(rowInSector, padNumber));
125  const PadCentre& padCentre = mapper.padCentre(pad);
126  const float localY = padCentre.Y() - (padY - padNumber - 0.5) * region.getPadWidth();
127  const float localYfactor = (cru.side() == Side::A) ? -1.f : 1.f;
128  float zPositionAbs = cluster.getTimeMean()*elParam.getZBinWidth()*gasParam.getVdrift();
129  if (!mTrackingCAO2Interface->GetParamContinuous()) zPositionAbs = detParam.getTPClength() - zPositionAbs;
130 
131  Point2D<float> clusterPos(padCentre.X(), localY);
132 
133  // sanity checks
134  if (zPositionAbs < 0 || (!mTrackingCAO2Interface->GetParamContinuous() && zPositionAbs > detParam.getTPClength())) {
135  LOG(INFO) << "Removing cluster " << icluster << "/" << inputClustersArray->size() << " time: " << cluster.getTimeMean() << ", abs. z: " << zPositionAbs << "\n";
136  continue;
137  }
138 
139  hltCluster.fX = clusterPos.X();
140  hltCluster.fY = clusterPos.Y() * (localYfactor);
141  hltCluster.fZ = zPositionAbs * (-localYfactor);
142  hltCluster.fRow = rowInSector;
143  hltCluster.fAmp = cluster.getQmax();
144  if (inputClustersChain) {
145  clusterCache[nClustersConverted] = cluster;
146  hltCluster.fId = nClustersConverted;
147  } else {
148  hltCluster.fId = icluster;
149  }
150 
151  cd.SetNumberOfClusters(cd.NumberOfClusters() + 1);
152  nClustersConverted++;
153  }
154  }
155 
156  if (nClustersTotal != nClustersConverted) {
157  LOG(INFO) << "Passed " << nClustersConverted << " (out of " << nClustersTotal << ") clusters to CA tracker\n";
158  }
159 
160  retVal = mTrackingCAO2Interface->RunTracking(mClusterData, tracks, nTracks, trackClusterIDs);
161  if (retVal == 0)
162  {
163  for (int i = 0; i < nTracks; i++) {
164  if (!tracks[i].OK()) continue;
165  TrackTPC trackTPC(
166  tracks[i].GetParam().GetX(), tracks[i].GetAlpha(),
167  { tracks[i].GetParam().GetY(), tracks[i].GetParam().GetZ(), tracks[i].GetParam().GetSinPhi(),
168  tracks[i].GetParam().GetDzDs(), tracks[i].GetParam().GetQPt() },
169  { tracks[i].GetParam().GetCov(0), tracks[i].GetParam().GetCov(1), tracks[i].GetParam().GetCov(2),
170  tracks[i].GetParam().GetCov(3), tracks[i].GetParam().GetCov(4), tracks[i].GetParam().GetCov(5),
171  tracks[i].GetParam().GetCov(6), tracks[i].GetParam().GetCov(7), tracks[i].GetParam().GetCov(8),
172  tracks[i].GetParam().GetCov(9), tracks[i].GetParam().GetCov(10), tracks[i].GetParam().GetCov(11),
173  tracks[i].GetParam().GetCov(12), tracks[i].GetParam().GetCov(13), tracks[i].GetParam().GetCov(14) });
174  for (int j = 0; j < tracks[i].NClusters(); j++) {
175  if (inputClustersChain) {
176  trackTPC.addCluster(clusterCache[trackClusterIDs[tracks[i].FirstClusterRef() + j]]);
177  } else {
178  //trackTPC.addCluster(*(static_cast<Cluster*>(inputClustersArray->At(trackClusterIDs[tracks[i].FirstClusterRef() + j]))));
179  trackTPC.addCluster((*inputClustersArray)[trackClusterIDs[tracks[i].FirstClusterRef() + j]]);
180  }
181  }
182  outputTracks->push_back(trackTPC);
183  }
184  }
185  mTrackingCAO2Interface->Cleanup();
186  return (retVal);
187 }
